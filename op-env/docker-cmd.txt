docker network create op_net
--------------------------------------------------

mysql
--------------------------------------------------
docker run -d `
    --network op_net `
    --hostname mysql `
    --name mysql `
    -p 3306:3306 `
    -e MYSQL_ROOT_PASSWORD=root `
    mysql
--------------------------------------------------

redis
--------------------------------------------------
docker run -d `
    --network op_net `
    --hostname redis `
    --name redis `
    -p 6379:6379 `
    redis
--------------------------------------------------

rabbitmq
--------------------------------------------------
docker run -d `
    --network op_net `
    --hostname rabbitmq `
    --name rabbitmq `
    -p 5672:5672 -p 15672:15672 `
    rabbitmq:3-management
--------------------------------------------------

elasticsearch
--------------------------------------------------
docker run -d `
    --network op_net `
    --hostname elasticsearch `
    --name elasticsearch `
    -p 9200:9200 -p 9300:9300 `
    -e "discovery.type=single-node" `
    elasticsearch
